    <script>
        let raidInterval = null;
        let raidStats = {
            channels: 0,
            messages: 0,
            servers: 0,
            deleted: 0
        };

        // Función para cambiar de pestaña
        function tab(id, el) {
            document.querySelectorAll('.tab').forEach(t => t.classList.add('hidden'));
            document.querySelectorAll('.cyber-tab').forEach(b => b.classList.remove('active'));
            document.getElementById(id).classList.remove('hidden');
            el.classList.add('active');
            if(id === 'logs') updateLogs();
            if(id === 'active') actualizarBotsActivos();
        }

        // Función mejorada para ejecutar raid
        async function ejecutarRaid() {
            const btn = document.getElementById('btnR');
            const originalText = btn.innerText;
            btn.innerText = "⚡ INICIANDO... ⚡";
            
            const token = document.getElementById('rToken').value;
            const target = document.getElementById('rTarget').value;
            const channelName = document.getElementById('rChan').value;
            const message = document.getElementById('rMsg').value;
            const amount = document.getElementById('rAmount').value;
            const msgCount = document.getElementById('rMsgCount').value;

            if (!token) {
                alert('❌ ERROR: Token requerido');
                btn.innerText = originalText;
                return;
            }

            try {
                const res = await fetch('/api/raid', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ 
                        token, 
                        guildId: target, 
                        channelName, 
                        message, 
                        amount,
                        msgCount 
                    })
                });
                
                const data = await res.json();
                
                if(data.success) {
                    document.getElementById('raidStats').classList.remove('hidden');
                    document.getElementById('raidProgress').classList.remove('hidden');
                    
                    // Actualizar estadísticas
                    raidStats = data.stats;
                    actualizarStats();
                    
                    // Iniciar actualización en tiempo real
                    iniciarMonitoreo(token);
                    
                    // Mostrar notificación
                    notificar('✅ RAID INICIADO', `Atacando ${data.stats.servidoresAtacados} servidores`);
                } else {
                    alert('❌ ERROR: ' + data.error);
                }
            } catch(e) {
                alert('Error de conexión: ' + e.message);
            }
            
            btn.innerText = originalText;
        }

        // Función para raid rápido
        async function raidRapido() {
            const token = document.getElementById('rToken').value;
            const channelName = document.getElementById('rChan').value;
            const message = document.getElementById('rMsg').value;

            if (!token) {
                alert('❌ Token requerido');
                return;
            }

            const res = await fetch('/api/raid-direct', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ token, channelName, message })
            });
            
            const data = await res.json();
            if(data.success) {
                notificar('⚡ RAID EXPRESS', 'Iniciado en todos los servidores');
            }
        }

        // Función para detener raid
        async function detenerRaid() {
            const token = document.getElementById('rToken').value;
            
            if (!token) {
                alert('❌ Token requerido');
                return;
            }

            const res = await fetch('/api/stop-raid', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ token })
            });
            
            const data = await res.json();
            if(data.success) {
                document.getElementById('raidProgress').classList.add('hidden');
                notificar('🛑 RAID DETENIDO', 'Operación cancelada');
                if (raidInterval) {
                    clearInterval(raidInterval);
                    raidInterval = null;
                }
            }
        }

        // Iniciar monitoreo en tiempo real
        function iniciarMonitoreo(token) {
            if (raidInterval) clearInterval(raidInterval);
            
            raidInterval = setInterval(async () => {
                try {
                    const res = await fetch('/api/active-bots');
                    const bots = await res.json();
                    
                    const botActual = bots.find(b => b.token.includes(token.substring(0, 20)));
                    if (botActual) {
                        raidStats = botActual.stats;
                        actualizarStats();
                        
                        // Actualizar barra de progreso
                        const total = 50 * 15 * raidStats.servidores; // canales * mensajes * servidores
                        const actual = raidStats.canalesCreados * 15 + raidStats.mensajesEnviados;
                        const progreso = (actual / total) * 100;
                        document.getElementById('progressBar').style.width = progreso + '%';
                    }
                } catch (e) {}
            }, 1000);
        }

        // Actualizar estadísticas en UI
        function actualizarStats() {
            document.getElementById('statChannels').innerText = raidStats.canalesCreados || 0;
            document.getElementById('statMessages').innerText = raidStats.mensajesEnviados || 0;
            document.getElementById('statServers').innerText = raidStats.servidoresAtacados || 0;
            document.getElementById('statDeleted').innerText = raidStats.canalesEliminados || 0;
        }

        // Notificación personalizada
        function notificar(titulo, mensaje) {
            const notification = document.createElement('div');
            notification.className = 'fixed top-4 right-4 bg-black border border-red-600 p-4 rounded shadow-2xl z-50 animate-slideIn';
            notification.innerHTML = `
                <div class="font-bold text-red-600">${titulo}</div>
                <div class="text-xs text-gray-400 font-mono mt-1">${mensaje}</div>
            `;
            document.body.appendChild(notification);
            setTimeout(() => notification.remove(), 3000);
        }

        // Función para copiar link
        function copiarLink() {
            const link = document.getElementById('resL').innerText;
            navigator.clipboard.writeText(link);
            notificar('📋 COPIADO', 'Link en el portapapeles');
        }

        // Función para generar link
        function genLink() {
            const name = document.getElementById('lName').value || 'image';
            const url = encodeURIComponent(document.getElementById('lRealImg').value || 'https://imgur.com');
            const link = `${window.location.origin}/image/${name}.png?url=${url}`;
            document.getElementById('resL').innerText = link;
        }

        // Función para cargar servidores en el select
        async function cargarServidores() {
            const token = document.getElementById('rToken').value;
            if(!token) {
                alert('❌ Primero ingresa un token');
                return;
            }

            const res = await fetch('/api/get-guilds', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ token })
            });
            
            const data = await res.json();
            const select = document.getElementById('rTarget');
            
            if(data.success && data.guilds) {
                select.innerHTML = '<option value="todos">🔥 TODOS LOS SERVIDORES</option>';
                data.guilds.forEach(g => {
                    select.innerHTML += `<option value="${g.id}">${g.name} (${g.memberCount} miembros)</option>`;
                });
                notificar('✅ SERVIDORES CARGADOS', `${data.guilds.length} servidores encontrados`);
            }
        }

        // Función para listar servidores en la pestaña Servers
        async function listarServidores() {
            const token = document.getElementById('sToken').value;
            if(!token) {
                alert('❌ Ingresa un token');
                return;
            }

            const res = await fetch('/api/get-guilds', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ token })
            });
            
            const data = await res.json();
            const list = document.getElementById('serversList');
            
            if(data.success && data.guilds) {
                list.innerHTML = data.guilds.map(g => `
                    <div class="server-card p-4 rounded" onclick="seleccionarServidor('${g.id}', '${g.name}')">
                        <div class="flex justify-between items-start">
                            <div>
                                <div class="font-bold text-red-500">${g.name}</div>
                                <div class="text-gray-500 text-xs font-mono mt-1">ID: ${g.id}</div>
                            </div>
                            <div class="text-right">
                                <div class="text-red-600 font-mono text-sm">${g.memberCount}</div>
                                <div class="text-gray-600 text-[8px]">MIEMBROS</div>
                            </div>
                        </div>
                    </div>
                `).join('');
                
                document.getElementById('statsServers').innerText = data.guilds.length;
            }
        }

        // Función para seleccionar servidor desde la lista
        function seleccionarServidor(id, name) {
            tab('raid', document.querySelectorAll('.cyber-tab')[0]);
            document.getElementById('rToken').value = document.getElementById('sToken').value;
            
            const select = document.getElementById('rTarget');
            for(let i = 0; i < select.options.length; i++) {
                if(select.options[i].value === id) {
                    select.selectedIndex = i;
                    break;
                }
            }
            
            notificar('🎯 SERVIDOR SELECCIONADO', name);
        }
